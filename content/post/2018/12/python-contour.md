import numpy as np
import matplotlib.pyplot as plt

# データの用意
data1 = np.array([
[34.4, 32.1, 30.5, 29.5, 29.1, 28, 30.4, 32.4, 34.8],
[36.3, 34.1, 32.5, 31.5, 31.2, 31.6, 32.7, 34.6, 36.9],
[37.8, 35.6, 32.8, 32.9, 32.7, 33.1, 34.3, 36.2, 38.7],
[38.6, 36.2, 34.7, 33.7, 33.5, 33.9, 35.2, 37.1, 39.7],
[38.6, 36.2, 34.6, 33.8, 33.6, 33.5, 35.4, 37.4, 40],
[37.9, 35.6, 33.9, 33.1, 33, 33.5, 34.9, 36.8, 39.5],
[36.4, 34.1, 32.6, 31.8, 31.8, 32.4, 33.6, 35.7, 38.1],
[33.9, 31.7, 30.4, 29.8, 29.9, 30.5, 31.7, 33.5, 36.1],
[30.7, 28.6, 27.5, 27.2, 27.5, 28.1, 29.4, 31, 33.4]
])

data2 = np.array([
[34, 31.8, 30.1, 29.2, 28.8, 28.7, 30.4, 32.3, 34.7],
[36, 33.8, 32.2, 31.2, 31, 30.7, 32.7, 34.2, 36.9],
[37.6, 35.4, 33.1, 32.7, 32.4, 32.1, 34.1, 35.9, 38.4],
[38.4, 36, 34.2, 33.4, 33.3, 33.7, 35, 37.1, 39.5],
[38.7, 36.1, 34.3, 33.4, 33.4, 34, 35.1, 37.1, 39.5],
[37.6, 35.3, 33.7, 32.9, 32.9, 33.5, 34.8, 36.8, 39.3],
[36.1, 33.9, 32.3, 31.7, 31.7, 32.3, 33.6, 35.5, 37.9],
[33.5, 31.6, 30.2, 29.6, 29.8, 30.5, 31.8, 33.6, 36],
[30.4, 28.4, 27.4, 27.2, 27.5, 28.2, 29.4, 31.1, 33.5]
])
cb_min = 25 # カラーバーの下端
cb_max = 40 # カラーバーの上端
data = data1 # 描画するデータの指定
X,Y=np.meshgrid(np.linspace(1, 9, 9),np.linspace(1, 9, 9)) # メッシュの用意
plt.figure(figsize=(8,6.2)) # 図の縦横比を指定する
div=15.0 # 図を描くのに何色用いるか
delta=(cb_max-cb_min)/div
interval=np.arange(cb_min,abs(cb_max)*2+delta,delta)[0:int(div)+1]
plt.contourf(X,Y,data,interval)
plt.colorbar(shrink=0.8) # カラーバーの長さを指定
plt.xlabel('The x coordinate of the measurement point') # x軸のラベル
plt.ylabel('The y coordinate of the measurement point') # y軸のラベル
plt.ylim(9,1) # y軸を反転させる(反転させた目盛りで上書き)
plt.show()
